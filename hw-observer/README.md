```
$ sudo apt -y install python3-pip
$ sudo pip3 install virtualenv
$ virtualenv .
$ pip3 install -r requirements.txt
```

```
$ pip3 install hoge
$ pip3 list
$ pip3 freeze > requirements.txt
```
